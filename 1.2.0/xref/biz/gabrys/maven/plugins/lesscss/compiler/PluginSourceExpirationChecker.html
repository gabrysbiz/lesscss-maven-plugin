13  package biz.gabrys.maven.plugins.lesscss.compiler;
14  
15  import java.util.Date;
16  
17  import org.apache.maven.plugin.logging.Log;
18  
19  import biz.gabrys.lesscss.extended.compiler.cache.FullCache;
20  import biz.gabrys.lesscss.extended.compiler.cache.SourceModificationDateCache;
21  import biz.gabrys.lesscss.extended.compiler.control.expiration.CompiledSourceExpirationChecker;
22  import biz.gabrys.lesscss.extended.compiler.control.expiration.CompiledSourceExpirationCheckerImpl;
23  import biz.gabrys.lesscss.extended.compiler.control.expiration.SourceExpirationChecker;
24  import biz.gabrys.lesscss.extended.compiler.control.expiration.SourceModificationDateBasedExpirationChecker;
25  import biz.gabrys.lesscss.extended.compiler.source.LessSource;
26  import biz.gabrys.lesscss.extended.compiler.source.SourceFactory;
27  
28  public class PluginSourceExpirationChecker implements CompiledSourceExpirationChecker {
29  
30      private final CompiledSourceExpirationChecker expirationChecker;
31      private final Log logger;
32  
33      public PluginSourceExpirationChecker(final FullCache cache, final SourceFactory sourceFactory, final Log logger) {
34          expirationChecker = new CompiledSourceExpirationCheckerWithLogger(cache, sourceFactory, logger);
35          this.logger = logger;
36      }
37  
38      public boolean isExpired(final LessSource source, final Date compilationDate) {
39          if (logger.isDebugEnabled()) {
40              logger.debug("Verificating whether the source file or at least one import need compilation...");
41          }
42          final boolean expired = expirationChecker.isExpired(source, compilationDate);
43          if (logger.isDebugEnabled() && !expired) {
44              logger.debug("Cache for the source file and all imports are up to date");
45          }
46          return expired;
47      }
48  
49      private static final class CompiledSourceExpirationCheckerWithLogger extends CompiledSourceExpirationCheckerImpl {
50  
51          private final Log logger;
52  
53          private CompiledSourceExpirationCheckerWithLogger(final FullCache cache, final SourceFactory sourceFactory, final Log logger) {
54              super(new SourceExpirationCheckerWithLogger(cache, logger), cache, cache, sourceFactory);
55              this.logger = logger;
56          }
57  
58          @Override
59          protected boolean isModifiedAfterLastCompilation(final LessSource source, final Date lastCompilationDate) {
60              final boolean expired = super.isModifiedAfterLastCompilation(source, lastCompilationDate);
61              if (logger.isDebugEnabled() && expired) {
62                  logger.debug(String.format("Cache for source in older than import %s, file need compilation", source.getPath()));
63              }
64              return expired;
65          }
66      }
67  
68      private static final class SourceExpirationCheckerWithLogger implements SourceExpirationChecker {
69  
70          private final SourceExpirationChecker expirationChecker;
71          private final Log logger;
72  
73          private SourceExpirationCheckerWithLogger(final SourceModificationDateCache cache, final Log logger) {
74              expirationChecker = new SourceModificationDateBasedExpirationChecker(cache);
75              this.logger = logger;
76          }
77  
78          public boolean isExpired(final LessSource source) {
79              final boolean expired = expirationChecker.isExpired(source);
80              if (logger.isDebugEnabled() && expired) {
81                  logger.debug(String.format("Cache for source file %s expired, file need compilation", source.getPath()));
82              }
83              return expired;
84          }
85      }
86  
87  }
