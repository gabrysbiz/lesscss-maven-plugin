Multiple source/output directories

If you want to use more than one source or output directory, then you must to configure multiple executions. 

<project>
   ...
   <build>
      <plugins>
         ...
         <plugin>
            <groupId>biz.gabrys.maven.plugins</groupId>
            <artifactId>lesscss-maven-plugin</artifactId>
            <version>1.2.0-SNAPSHOT</version>
            <executions>
               <execution>
                  <id>lesscss-common</id>
                  <goals>
                     <goal>compile</goal>
                  </goals>
                  <configuration>
                     <sourceDirectory>src/main/less/common</sourceDirectory>
                     <outputDirectory>target/css/common</outputDirectory>
                  </configuration>
               </execution>
               <execution>
                  <id>lesscss-vendors</id>
                  <goals>
                     <goal>compile</goal>
                  </goals>
                  <configuration>
                     <sourceDirectory>src/vendors</sourceDirectory>
                     <outputDirectory>target/css/vendors</outputDirectory>
                  </configuration>
               </execution>
               ...
            </executions>
         </plugin>
      </plugins>
   </build>
