13  package biz.gabrys.maven.plugins.lesscss;
14  
15  import java.io.File;
16  import java.io.IOException;
17  import java.util.LinkedHashMap;
18  import java.util.Map;
19  
20  import org.apache.commons.io.FileUtils;
21  
22  import biz.gabrys.lesscss.compiler2.FileSystemOptionsBuilder;
23  import biz.gabrys.lesscss.compiler2.LessOptions;
24  import biz.gabrys.lesscss.compiler2.LessOptionsBuilder;
25  import biz.gabrys.lesscss.compiler2.LessVariableOptionsBuilder;
26  import biz.gabrys.lesscss.compiler2.LineNumbersValue;
27  
28  public class Options {
29  
30      protected boolean silent;
31      protected boolean strictImports;
32      protected boolean compress;
33      protected boolean ieCompatibility = true;
34      protected boolean javaScript = true;
35      protected String[] includePaths = new String[0];
36      protected String lineNumbers = LineNumbersValue.OFF.getValue();
37      protected String rootPath;
38      protected boolean relativeUrls;
39      protected boolean strictMath;
40      protected boolean strictUnits;
41  
42      protected String sourceMapRootPath;
43      protected String sourceMapBasePath;
44      protected boolean sourceMapLessInline;
45      protected String sourceMapUrl;
46  
47      protected Banner banner = new Banner();
48      protected Map<String, String> globalVariables = new LinkedHashMap<String, String>();
49      protected Map<String, String> modifyVariables = new LinkedHashMap<String, String>();
50  
51      protected boolean httpEnabled = true;
52      protected boolean ftpEnabled = true;
53      protected boolean classpathEnabled = true;
54      protected String classpathLoadedDependenciesTypes = "jar,war";
55  
56      public LessOptions toLessOptions(final String encoding) throws IOException {
57          final FileSystemOptionsBuilder fileSystemOptionsBuilder = new FileSystemOptionsBuilder();
58          if (httpEnabled) {
59              fileSystemOptionsBuilder.appendHttp();
60          }
61          if (ftpEnabled) {
62              fileSystemOptionsBuilder.appendFtp();
63          }
64          if (classpathEnabled) {
65              fileSystemOptionsBuilder.appendClassPath();
66          }
67          fileSystemOptionsBuilder.appendLocal();
68  
69          return new LessOptionsBuilder() //
70                  .silent(silent) //
71                  .strictImports(strictImports) //
72                  .compress(compress) //
73                  .ieCompatibility(ieCompatibility) //
74                  .javaScript(javaScript) //
75                  .includePaths(includePaths) //
76                  .lineNumbers(LineNumbersValue.toLineNumbersValue(lineNumbers)) //
77                  .rootPath(rootPath) //
78                  .relativeUrls(relativeUrls) //
79                  .strictMath(strictMath) //
80                  .strictUnits(strictUnits) //
81                  .sourceMapRootPath(sourceMapRootPath) //
82                  .sourceMapBasePath(sourceMapBasePath) //
83                  .sourceMapLessInline(sourceMapLessInline) //
84                  .sourceMapUrl(sourceMapUrl) //
85                  .encoding(encoding) //
86                  .fileSystems(fileSystemOptionsBuilder.build()) //
87                  .banner(bannerToString(encoding)) //
88                  .globalVariables(new LessVariableOptionsBuilder().append(globalVariables).build()) //
89                  .modifyVariables(new LessVariableOptionsBuilder().append(modifyVariables).build()) //
90                  .build();
91      }
92  
93      private String bannerToString(final String encoding) throws IOException {
94          if (banner.text != null) {
95              return banner.text;
96          }
97          if (banner.file != null) {
98              return FileUtils.readFileToString(banner.file, encoding);
99          }
100         return null;
101     }
102 
103     public static class Banner {
104         protected File file;
105         protected String text;
106     }
107 }
