13  package biz.gabrys.maven.plugins.lesscss.compiler;
14  
15  import java.util.Date;
16  
17  import org.apache.maven.plugin.MojoFailureException;
18  
19  import biz.gabrys.lesscss.compiler.CompilerException;
20  import biz.gabrys.lesscss.compiler.CompilerOptions;
21  import biz.gabrys.lesscss.extended.compiler.ExtendedCompiler;
22  import biz.gabrys.lesscss.extended.compiler.ExtendedCompilerException;
23  import biz.gabrys.lesscss.extended.compiler.cache.CompilationDateCache;
24  import biz.gabrys.lesscss.extended.compiler.source.LessSource;
25  
26  public class PluginCompiler {
27  
28      private final ExtendedCompiler compiler;
29      private final CompilationDateCache cache;
30      private Date compilationDate;
31  
32      public PluginCompiler(final ExtendedCompiler compiler, final CompilationDateCache cache) {
33          this.cache = cache;
34          this.compiler = compiler;
35      }
36  
37      public String compile(final LessSource source, final CompilerOptions options) throws MojoFailureException {
38          compilationDate = new Date();
39          String compiledCode;
40          try {
41              compiledCode = compiler.compile(source, options);
42          } catch (final ExtendedCompilerException e) {
43              throw new MojoFailureException(e.getMessage(), e);
44          } catch (final CompilerException e) {
45              throw new MojoFailureException(e.getMessage(), e);
46          }
47  
48          if (cache != null && cache.hasCompilationDate(source)) {
49              compilationDate = cache.getCompilationDate(source);
50          }
51          return compiledCode;
52      }
53  
54      public Date getCompilationDate() {
55          return (Date) compilationDate.clone();
56      }
57  }
