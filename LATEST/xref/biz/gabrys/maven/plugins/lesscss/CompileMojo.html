151     /**
152      * Defines <a href="http://lesscss.org/usage/index.html#less-options">Less compiler options</a> responsible for
153      * controlling the compilation process.<br>
154      * Base options:
155      * <ul>
156      * <li><b>compress</b> - whether a CSS code should be compressed (default: <code>false</code>)</li>
157      * <li><b>ieCompatibility</b> - whether a CSS code should be compatible with Internet Explorer browser (default:
158      * <code>true</code>)</li>
159      * <li><b>includePaths</b> - available include paths (default: <code>[]</code>)</li>
160      * <li><b>javaScript</b> - whether a compiler should allow usage of JavaScript language (default:
161      * <code>true</code>)</li>
162      * <li><b>lineNumbers</b> - whether a compiler should generate inline source-mapping:
163      * <ul>
164      * <li><i>empty</i> - line numbers won't be put in CSS code (default)</li>
165      * <li><b>comments</b> - line numbers will be put in CSS comments blocks</li>
166      * <li><b>mediaquery</b> - line numbers will be put in &#64;media queries</li>
167      * <li><b>all</b> - line numbers will be put in CSS comments blocks and &#64;media queries</li>
168      * </ul>
169      * </li>
170      * <li><b>relativeUrls</b> - whether a compiler should rewrite relative URLs (default: <code>false</code>)</li>
171      * <li><b>rootPath</b> - a path which will be added to every generated import and URL in CSS code (default:
172      * <code>null</code>)</li>
173      * <li><b>silent</b> - whether a compiler shouldn't log compilation warnings (default: <code>false</code>)</li>
174      * <li><b>strictImports</b> - whether a compiler should disallow an @import operation inside of either @media blocks
175      * or other selector blocks (default: <code>false</code>)</li>
176      * <li><b>strictMath</b> - whether a compiler should try and process all maths in Less code (default:
177      * <code>false</code>)</li>
178      * <li><b>strictUnits</b> - whether a compiler should guess units in Less code when it does maths (default:
179      * <code>false</code>)</li>
180      * </ul>
181      * Source Map options:
182      * <ul>
183      * <li><b>sourceMapBasePath</b> - a path that will be removed from each of the Less file paths inside the Source Map
184      * and also from the path to the map file specified in your output CSS (default: <code>null</code>)</li>
185      * <li><b>sourceMapLessInline</b> - whether a compiler should include all of the Less files in to the Source Map
186      * (default: <code>false</code>)</li>
187      * <li><b>sourceMapRootPath</b> - a path that will be prepended to each of the Less file paths inside the Source Map
188      * and also to the path to the map file specified in your output CSS (default: <code>null</code>)</li>
189      * <li><b>sourceMapUrl</b> - a path which will overwrite the URL in the CSS that points at the Source Map file
190      * (default: <code>null</code>)</li>
191      * </ul>
192      * Additional options:
193      * <ul>
194      * <li><b>banner</b> - a banner which will be inserted to source files before the compilation (default:
195      * <code>null</code>):
196      * <ul>
197      * <li><b>file</b> - loads banner form the specified file (ignored when <b>text</b> is set)</li>
198      * <li><b>text</b> - uses the specified text</li>
199      * </ul>
200      * </li>
201      * <li><b>globalVariables</b> - variables that can be referenced by the files (default: <code>{}</code>)</li>
202      * <li><b>modifyVariables</b> - variables that can overwrite variables defined in the files (default:
203      * <code>{}</code>)</li>
204      * </ul>
205      * Non-standard options:
206      * <ul>
207      * <li><b>httpEnabled</b> - whether the plugin allows using <code>http://</code> protocol (default
208      * <code>true</code>)</li>
209      * <li><b>ftpEnabled</b> - whether the plugin allows using <code>ftp://</code> protocol (default
210      * <code>true</code>)</li>
211      * <li><b>classpathEnabled</b> - whether the plugin allows using <code>classpath://</code> protocol (default
212      * <code>true</code>)</li>
213      * <li><b>classpathLoadedDependenciesTypes</b> - a comma separated types list of dependencies whose will be added to
214      * the plugin class path (default: <code>jar,war</code>)</li>
215      * </ul>
216      * @since 2.0.0
217      */

262     private void logParameters() {
263         if (!getLog().isDebugEnabled()) {
264             return;
265         }
266 
267         final ParametersLogBuilder logger = new ParametersLogBuilder(getLog());
268         logger.append("skip", skip);
269         logger.append("verbose", verbose, new SimpleSanitizer(verbose, Boolean.TRUE));
270         // logger.append("force", force, new SimpleSanitizer(!watch || !force, Boolean.FALSE));
271         // logger.append("alwaysOverwrite", alwaysOverwrite, new SimpleSanitizer(!(!watch && force && !alwaysOverwrite),
272         // Boolean.TRUE));
273         logger.append("sourceDirectory", sourceDirectory);
274         logger.append("outputDirectory", outputDirectory);
275         logger.append("filesetPatternFormat", filesetPatternFormat);
276         logger.append("includes", includes, new LazySimpleSanitizer(includes.length != 0, new ValueContainer() {
277 
278             @Override
279             public Object getValue() {
280                 return getDefaultIncludes();
281             }
282         }));
283         logger.append("excludes", excludes);
284         logger.append("options.silent", options.silent);
285         logger.append("options.strictImports", options.strictImports);
286         logger.append("options.compress", options.compress);
287         logger.append("options.ieCompatibility", options.ieCompatibility);
288         logger.append("options.javaScript", options.javaScript);
289         logger.append("options.includePaths", options.includePaths);
290         logger.append("options.lineNumbers", options.lineNumbers);
291         logger.append("options.rootPath", options.rootPath);
292         logger.append("options.relativeUrls", options.relativeUrls);
293         logger.append("options.strictMath", options.strictMath);
294         logger.append("options.strictUnits", options.strictUnits);
295         logger.append("options.sourceMapRootPath", options.sourceMapRootPath);
296         logger.append("options.sourceMapBasePath", options.sourceMapBasePath);
297         logger.append("options.sourceMapLessInline", options.sourceMapLessInline);
298         logger.append("options.sourceMapUrl", options.sourceMapUrl);
299         logger.append("options.banner.file", options.banner.file);
300         logger.append("options.banner.text", options.banner.text);
301         logger.append("options.globalVariables", options.globalVariables);
302         logger.append("options.modifyVariables", options.modifyVariables);
303         logger.append("options.httpEnabled", options.httpEnabled);
304         logger.append("options.ftpEnabled", options.ftpEnabled);
305         logger.append("options.classpathEnabled", options.classpathEnabled);
306         logger.append("options.classpathLoadedDependenciesTypes", options.classpathLoadedDependenciesTypes);
307         logger.append("encoding", encoding);
308         logger.append("outputFileFormat", outputFileFormat);
309         logger.append("watch", watch);
310         logger.append("watchInterval", watchInterval, new ValueToStringConverter() {
311 
312             private static final long MILLISECONDS_IN_SECOND = 1000L;
313 
314             @Override
315             public String convert(final Object value) {
316                 final Integer number = (Integer) value;
317                 final StringBuilder text = new StringBuilder();
318                 text.append(number);
319                 if (number > 0) {
320                     text.append(" (");
321                     text.append(new TimeSpan(number * MILLISECONDS_IN_SECOND));
322                     text.append(')');
323                 }
324                 return text.toString();
325             }
326         }, new SimpleSanitizer(watchInterval > 0, Integer.valueOf(1)));
327         logger.append("workingDirectory", workingDirectory);
328         logger.debug();
329     }

331     private String[] getDefaultIncludes() {
332         if (ScannerPatternFormat.ANT.name().equalsIgnoreCase(filesetPatternFormat)) {
333             return new String[] { "**/*.less" };
334         } else {
335             return new String[] { "^.+\\.less$" };
336         }
337     }
338 
339     private void calculateParameters() {
340         if (getLog().isDebugEnabled()) {
341             verbose = true;
342         }
343         if (includes.length == 0) {
344             includes = getDefaultIncludes();
345         }
346         if (watchInterval < 1) {
347             watchInterval = 1;
348         }
349     }

351     @Override
352     public void execute() throws MojoFailureException {
353         logParameters();
354         if (skip) {
355             getLog().info("Skips job execution");
356             return;
357         }
358         calculateParameters();
359 
360         if (options.classpathEnabled) {
361             addDependenciesToClasspath();
362         }
363 
364         if (watch) {
365             runWatchMode();
366         } else {
367             runCompilation();
368         }
369     }
370 
371     private void addDependenciesToClasspath() {
372         if (verbose) {
373             getLog().info("Adding project dependencies to classpath...");
374         }
375         final ContextClassLoaderExtender extender = new ContextClassLoaderExtender(project, getLog());
376         final String[] types = options.classpathLoadedDependenciesTypes.split(",");
377         for (int i = 0; i < types.length; ++i) {
378             types[i] = types[i].trim();
379         }
380         extender.addDependencies(types);
381     }
382 
383     private void runWatchMode() throws MojoFailureException {
384         getLog().info("Starts watch mode on the " + sourceDirectory);
385         Thread.currentThread().setPriority(Thread.MIN_PRIORITY);
386         final long interval = watchInterval * 1000L;
387         while (true) {
388             runCompilation();
389             try {
390                 Thread.sleep(interval);
391             } catch (final InterruptedException e) {
392                 Thread.currentThread().interrupt();
393                 return;
394             }
395         }
396     }
397 
398     private void runCompilation() throws MojoFailureException {
399         if (!sourceDirectory.exists()) {
400             getLog().warn("Source directory does not exist: " + sourceDirectory.getAbsolutePath());
401             return;
402         }
403         final Collection<File> files = getFiles();
404         if (files.isEmpty()) {
405             getLog().warn("No sources to compile");
406             return;
407         }
408         compileFiles(files);
409     }
410 
411     private Collection<File> getFiles() {
412         final ScannerPatternFormat patternFormat = ScannerPatternFormat.toPattern(filesetPatternFormat);
413         final FileScanner scanner = new ScannerFactory().create(patternFormat, getLog());
414         if (verbose) {
415             getLog().info("Scanning directory for sources...");
416         }
417         return scanner.getFiles(sourceDirectory, includes, excludes);
418     }
419 
420     private void compileFiles(final Collection<File> files) throws MojoFailureException {
421         final String sourceFilesText = "source" + (files.size() != 1 ? "s" : "");
422         getLog().info(String.format("Compiling %s %s to %s", files.size(), sourceFilesText, outputDirectory.getAbsolutePath()));
423 
424         final LessCompiler compiler = new LessCompiler();
425         final Timer timer = SystemTimer.getStartedTimer();
426         final LessOptions lessOptions = createOptions();
427         for (final File file : files) {
428             compileFile(compiler, file, lessOptions);
429         }
430 
431         getLog().info(String.format("Finished %s compilation in %s", sourceFilesText, timer.stop()));
432     }
433 
434     private LessOptions createOptions() throws MojoFailureException {
435         try {
436             return options.toLessOptions(encoding);
437         } catch (final Exception e) {
438             throw new MojoFailureException("Cannot prepare compiler configuration", e);
439         }
440     }
441 
442     private void compileFile(final LessCompiler compiler, final File source, final LessOptions options) throws MojoFailureException {
443         Timer timer = null;
444         if (verbose) {
445             getLog().info("Processing Less source: " + source.getAbsolutePath());
446             timer = SystemTimer.getStartedTimer();
447         }
448         final File destination = new DestinationFileCreator(sourceDirectory, outputDirectory, outputFileFormat).create(source);
449         compiler.compile(source, destination, options);
450         if (timer != null) {
451             getLog().info("Finished in " + timer.stop());
452         }
453     }
