68  /**

70   * using <a href="http://lesscss-extended-compiler.projects.gabrys.biz/">extended version</a> of the
71   * <a href="http://lesscss-compiler.projects.gabrys.biz/">LessCSS Compiler</a>.
72   * @since 1.0
73   */

260     private void logParameters() {
261         if (!getLog().isDebugEnabled()) {
262             return;
263         }
264 
265         final ParametersLogBuilder logger = new ParametersLogBuilder(getLog());
266         logger.append("skip", skip);
267         logger.append("verbose", verbose, new SimpleSanitizer(verbose, Boolean.TRUE));
268         logger.append("force", force, new SimpleSanitizer(!watch || watch && !force, Boolean.FALSE));
269         logger.append("alwaysOverwrite", alwaysOverwrite, new SimpleSanitizer(!(!watch && force && !alwaysOverwrite), Boolean.TRUE));
270         logger.append("sourceDirectory", sourceDirectory);
271         logger.append("outputDirectory", outputDirectory);
272         logger.append("filesetPatternFormat", filesetPatternFormat);
273         logger.append("includes", includes, new LazySimpleSanitizer(includes.length != 0, new ValueContainer() {
274 
275             public Object getValue() {
276                 return getDefaultIncludes();
277             }
278         }));
279         logger.append("excludes", excludes);
280         logger.append("compilerType", compilerType);
281         logger.append("classpathLoadedDependenciesTypes", classpathLoadedDependenciesTypes,
282                 new LazySimpleSanitizer(classpathLoadedDependenciesTypes.length != 0, new ValueContainer() {
283 
284                     public Object getValue() {
285                         return getDefaultClasspathLoadedDependenciesTypes();
286                     }
287                 }));
288         logger.append("addCommentsWithPaths", addCommentsWithPaths, new SimpleSanitizer(!compress, Boolean.FALSE));
289         logger.append("addCommentsWithPathsClassPrefix", addCommentsWithPathsClassPrefix);
290         logger.append("compress", compress);
291         logger.append("compilerOptions", compilerOptions);
292         logger.append("encoding", encoding);
293         logger.append("outputFileFormat", outputFileFormat);
294         logger.append("watch", watch);
295         logger.append("watchInterval", watchInterval, new ValueToStringConverter() {
296 
297             private static final long MILLISECONDS_IN_SECOND = 1000L;
298 
299             public String convert(final Object value) {
300                 final Integer number = (Integer) value;
301                 final StringBuilder text = new StringBuilder();
302                 text.append(number);
303                 if (number > 0) {
304                     text.append(" (");
305                     text.append(new Time(number * MILLISECONDS_IN_SECOND));
306                     text.append(')');
307                 }
308                 return text.toString();
309             }
310         }, new SimpleSanitizer(watchInterval > 0, Integer.valueOf(1)));
311         logger.append("workingDirectory", workingDirectory);
312         logger.debug();
313     }

315     private String[] getDefaultIncludes() {
316         if (ScannerPatternFormat.ANT.name().equalsIgnoreCase(filesetPatternFormat)) {
317             return new String[] { "**/*.less" };
318         } else {
319             return new String[] { "^.+\\.less$" };
320         }
321     }

323     private static String[] getDefaultClasspathLoadedDependenciesTypes() {
324         return new String[] { "jar", "war", "zip" };
325     }

327     private void calculateParameters() {
328         if (getLog().isDebugEnabled()) {
329             verbose = true;
330         }
331         if (watch) {
332             force = false;
333         }
334         if (force) {
335             alwaysOverwrite = true;
336         }
337         if (compress) {
338             addCommentsWithPaths = false;
339         }
340         if (includes.length == 0) {
341             includes = getDefaultIncludes();
342         }
343         if (classpathLoadedDependenciesTypes.length == 0) {
344             classpathLoadedDependenciesTypes = getDefaultClasspathLoadedDependenciesTypes();
345         }
346         if (watchInterval < 1) {
347             watchInterval = 1;
348         }
349     }

351     public void execute() throws MojoExecutionException, MojoFailureException {
352         logParameters();
353         if (skip) {
354             getLog().info("Skips job execution");
355             return;
356         }
357         calculateParameters();
358 
359         if ("full".equals(compilerType)) {
360             addDependenciesToClasspath();
361         }
362 
363         if (watch) {
364             runWatchMode();
365         } else {
366             runCompilation();
367         }
368     }

370     private void addDependenciesToClasspath() throws MojoExecutionException {
371         if (verbose) {
372             getLog().info("Adding project dependencies to classpath...");
373         }
374         final ContextClassLoaderExtender extender = new ContextClassLoaderExtender(project, getLog());
375         extender.addDependencies(classpathLoadedDependenciesTypes);
376     }

378     private void runWatchMode() throws MojoFailureException {
379         getLog().info("Starts watch mode on the " + sourceDirectory);
380         Thread.currentThread().setPriority(Thread.MIN_PRIORITY);
381         final long interval = watchInterval * 1000L;
382         while (true) {
383             runCompilation();
384             try {
385                 Thread.sleep(interval);
386             } catch (final InterruptedException e) {
387                 Thread.currentThread().interrupt();
388                 return;
389             }
390         }
391     }
392 
393     private void runCompilation() throws MojoFailureException {
394         if (!sourceDirectory.exists()) {
395             getLog().warn("Source directory does not exist: " + sourceDirectory.getAbsolutePath());
396             return;
397         }
398         final Collection<File> files = getFiles();
399         if (files.isEmpty()) {
400             getLog().warn("No sources to compile");
401             return;
402         }
403 
404         if (force) {
405             deleteWorkingDirectory();
406         }
407         compileFiles(files);
408     }
409 
410     private Collection<File> getFiles() {
411         final ScannerPatternFormat patternFormat = ScannerPatternFormat.toPattern(filesetPatternFormat);
412         final FileScanner scanner = new ScannerFactory().create(patternFormat, getLog());
413         if (verbose) {
414             getLog().info("Scanning directory for sources...");
415         }
416         return scanner.getFiles(sourceDirectory, includes, excludes);
417     }
418 
419     private void deleteWorkingDirectory() throws MojoFailureException {
420         if (!workingDirectory.exists()) {
421             return;
422         }
423         if (getLog().isDebugEnabled()) {
424             getLog().debug("Deleting working directory: " + workingDirectory.getAbsolutePath());
425         }
426         try {
427             FileUtils.deleteDirectory(workingDirectory);
428         } catch (final IOException e) {
429             throw new MojoFailureException(String.format("Cannot delete working directory: %s", workingDirectory.getAbsolutePath()), e);
430         }
431     }
432 
433     private void compileFiles(final Collection<File> files) throws MojoFailureException {
434         final PluginCompiler compiler = createCompiler();
435         final CompilerOptions options = createOptions();
436         final String sourceFilesText = "source" + (files.size() != 1 ? "s" : "");
437         getLog().info(String.format("Compiling %s %s to %s", files.size(), sourceFilesText, outputDirectory.getAbsolutePath()));
438         final Timer timer = SystemTimer.getStartedTimer();
439         for (final File file : files) {
440             compileFile(compiler, options, file);
441         }
442         getLog().info(String.format("Finished %s compilation in %s", sourceFilesText, timer.stop()));
443     }

445     private PluginCompiler createCompiler() {
446         if (getLog().isDebugEnabled()) {
447             getLog().debug("Creating compiler...");
448         }
449         FullCache cache = null;
450         ExtendedCompiler compiler;
451         SourceFactory sourceFactory;
452         if ("full".equalsIgnoreCase(compilerType)) {
453             cache = new FullCacheBuilder().withDirectory(workingDirectory).create();
454             sourceFactory = new SourceFactoryBuilder().withClasspath().withStandard().create();
455             PostCompilationProcessor postProcessor = null;
456             if (addCommentsWithPaths) {
457                 cache = new FullCacheAdapterBuilder(cache)
458                         .withSourceCodeCache(new PathInCommentSourceCodeCache(cache, addCommentsWithPathsClassPrefix)).create();
459                 postProcessor = new PathInCommentPostProcessor(addCommentsWithPathsClassPrefix);
460             }
461             compiler = new CachingSourceCodeExtendedCompilerBuilder(cache).withSourceFactory(sourceFactory).withPostProcessor(postProcessor)
462                     .create();
463         } else if ("local".equalsIgnoreCase(compilerType)) {
464             sourceFactory = force ? null : new SourceFactoryBuilder().withLocal().create();
465             compiler = new NonCachingExtendedCompilerBuilder().create();
466         } else {
467             throw new IllegalArgumentException(String.format("Cannot find compiler for type \"%s\"", compilerType));
468         }
469         if (verbose) {
470             compiler = new LoggingCompiler(compiler, getLog());
471         }
472         if (!force) {
473             cache = cache != null ? cache : new FullCacheBuilder().withDirectory(workingDirectory).create();
474             if (verbose) {
475                 final FullCacheAdapterBuilder builder = new FullCacheAdapterBuilder(cache);
476                 builder.withCompiledCodeCache(new LoggingCompiledCodeCache(cache, getLog()));
477                 if (getLog().isDebugEnabled()) {
478                     builder.withCompilationDateCache(new LoggingCompilationDateCache(cache, getLog()));
479                 }
480                 cache = builder.create();
481             }
482             final CompiledSourceExpirationChecker expirationChecker = new PluginSourceExpirationChecker(cache, sourceFactory, getLog());
483             compiler = new CachingCompiledCodeExtendedCompiler(compiler, expirationChecker, cache, cache);
484         }
485         return new PluginCompiler(compiler, cache);
486     }
487 
488     private CompilerOptions createOptions() {
489         final CompilerOptionsBuilder builder = new CompilerOptionsBuilder();
490         builder.setMinified(compress);
491         final CompilerOptions options = builder.create();
492         final List<Object> arguments = new ArrayList<Object>();
493         arguments.addAll(options.getArguments());
494         arguments.addAll(Arrays.asList(compilerOptions));
495         return new CompilerOptions(arguments);
496     }
497 
498     private void compileFile(final PluginCompiler compiler, final CompilerOptions options, final File source) throws MojoFailureException {
499         Timer timer = null;
500         if (verbose) {
501             getLog().info("Processing Less source: " + source.getAbsolutePath());
502             timer = SystemTimer.getStartedTimer();
503         }
504         final String compiled = compiler.compile(new LocalSource(source, encoding), options);
505         saveCompiledCode(source, compiled, compiler.getCompilationDate());
506         if (timer != null) {
507             getLog().info("Finished in " + timer.stop());
508         }
509     }
510 
511     private void saveCompiledCode(final File source, final String compiled, final Date compilationDate) throws MojoFailureException {
512         final File destination = new DestinationFileCreator(sourceDirectory, outputDirectory, outputFileFormat).create(source);
513 
514         final boolean skipsFileSaving = !force && !alwaysOverwrite && destination.exists()
515                 && compilationDate.before(new Date(destination.lastModified()));
516         if (skipsFileSaving) {
517             if (verbose) {
518                 getLog().info("Skips saving CSS compiled code to file, because cached version is older than destination file: "
519                         + destination.getAbsolutePath());
520             }
521             return;
522         }
523 
524         if (verbose) {
525             getLog().info("Saving CSS code to " + destination.getAbsolutePath());
526         }
527         try {
528             FileUtils.write(destination, compiled, encoding);
529         } catch (final IOException e) {
530             throw new MojoFailureException(String.format("Cannot save CSS compiled code to file: %s", destination.getAbsolutePath()), e);
531         }
532     }
